Get generated credentials

- BASIC AUTH
    export BASIC_AUTH_USER=$(kubectl get secret operation-creds -o jsonpath="{.data.basic-auth-user}" | base64 -d)
    export BASIC_AUTH_PASSWORD=$(kubectl get secret operation-creds -o jsonpath="{.data.basic-auth-password}" | base64 -d)
    echo "$BASIC_AUTH_USER:$BASIC_AUTH_PASSWORD"

- MINIO
    export MINIO_USER=$(kubectl get secret minio-creds -o jsonpath="{.data.root-user}" | base64 -d)
    export MINIO_PASSWORD=$(kubectl get secret minio-creds -o jsonpath="{.data.root-password}" | base64 -d)
    echo "$MINIO_USER:$MINIO_PASSWORD"

- AWS
    export AWS_ACCESS_KEY=$(kubectl get secret aws-creds -o jsonpath="{.data.aws_access_key_id}" | base64 -d)
    export AWS_SECRET_KEY=$(kubectl get secret aws-creds -o jsonpath="{.data.aws_secret_key}" | base64 -d)
    echo -n "AWS_ACCESS_KEY:$AWS_ACCESS_KEY\nAWS_SECRET_KEY: $AWS_SECRET_KEY"

- DASHBOARD
    export DASHBOARD_USER=$(kubectl get secret dashboard-creds -o jsonpath="{.data.dashboard-user}" | base64 -d)
    export DASHBOARD_PASS=$(kubectl get secret dashboard-creds -o jsonpath="{.data.dashboard-password}" | base64 -d)
    echo -n "$DASHBOARD_USER:$DASHBOARD_PASS"
